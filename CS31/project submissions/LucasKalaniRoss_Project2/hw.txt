This program prompts the user for a number, then prints a right triangle out of "#", where the length of the bottom side is the number that the user provided.

3.
#include <iostream>
using namespace std;
int main()
{
    int side;

    cout << "Enter a number: ";
    cin >> side;

    for (int i = 0; i < side; i++)
    {
	int j = i;
	while (j >= 0)
	{
	    cout << "#";
	    j--;
	}
	cout << "\n";
    }
}
